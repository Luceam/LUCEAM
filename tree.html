#include < stdio.h>
#include < stdlib.h>

typedef struct TreeNode
{
    int data;
    TreeNode *llink;
    TreeNode *rlink;
};

void preorder(TreeNode *tree) //전위순회 함수
{
    if(tree){

    printf("[%d] ", tree->data);

    preorder(tree->llink);

    preorder(tree->rlink);

    }

}

 
void inorder(TreeNode *tree) //중위순회 함수
{

    if(tree){

    inorder(tree->llink);

    printf("[%d] ", tree->data);

    inorder(tree->rlink);

    }
}

 
void postorder(TreeNode *tree) //후위순회 함수
{

    if(tree){

    postorder(tree->llink);

    postorder(tree->rlink);

    printf("[%d] ", tree->data);

    }
}

 
int get_node(TreeNode *tree)
{

    int cnt=0;

    if(tree!=NULL) 

        cnt = 1+ get_node(tree->llink) + get_node(tree->rlink);

    return cnt;
}

 
int max(int x, int y)
{

    if(x>y) return x;

    else return y;
}

 
int get_height(TreeNode *tree)
{

    int height=0;

    if(tree!=NULL)

        height=1+ max(get_height(tree->llink), get_height(tree->rlink));

    return height;


}

 
TreeNode *insert_node(TreeNode *tree, int item)
{

    TreeNode *p=tree , *node =(TreeNode *)malloc(sizeof(TreeNode));

    node->data=item;  node->llink=node->rlink=NULL;

    if(p==NULL) return node;

    if(p->data < node->data) //왼쪽

        p=insert_node(p->llink, item);

    else //오른쪽

        p=insert_node(p->rlink, item);

    return tree;

}
